// Add this to your main script to calculate the payment async function calculateOwnerPayout(carId) { const doc = await db.collection('fleet').doc(carId).get(); const data = doc.data(); // Math: Amount (CJ) - (Amount * Pct (CL)) const net = data.basePayment - (data.basePayment * (data.commissionPct / 100)); return { amount: net, nextDate: data.cycleDays // Days from CK }; }
